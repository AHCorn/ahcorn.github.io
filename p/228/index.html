其他用户

其实并没有多复杂，只需要加上泽泽之前写的获取缩略图就可以了

 1/** 输出文章缩略图 */ 
 2function showThumbnail($widget)
 3{ 
 4    // 当文章无图片时的默认缩略图
 5    $rand = rand(1,5); // 随机 1-5 张缩略图
 6    $random = $widget->widget('Widget_Options')->themeUrl . '/img/sj/' . $rand . '.jpg'; // 随机缩略图路径
 7   // $random = $widget->widget('Widget_Options')->themeUrl . '/img/mr.jpg'; // 若只想要一张默认缩略图请删除本行开头的"//"
 8
 9    $attach = $widget->attachments(1)->attachment;
10    $pattern = '/\<img.*?src\=\"(.*?)\"[^>]*>/i'; 
11    
12
13if (preg_match_all($pattern, $widget->content, $thumbUrl)) {
14         echo $thumbUrl[1][0];
15    } else     if ($attach->isImage) {
16      echo $attach->url; 
17    } else {
18        echo $random;
19    }
20}
